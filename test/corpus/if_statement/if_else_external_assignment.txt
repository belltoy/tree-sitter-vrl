==========================

# result: true

.a = 5
if true {
  .a = 0.0
} else {
  .a = "string"
}
assert!(.a == 0.0)
assert!(type_def(.a) == {"float": true, "bytes": true})

.a = 5
if false {
  .a = 0.0
} else {
  .a = "string"
}
assert!(.a == "string")
assert!(type_def(.a) == {"float": true, "bytes": true})
